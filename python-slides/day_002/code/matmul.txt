>>> I = np.identity(2); I
array([[1., 0.],
       [0., 1.]])
>>> A = np.array([[0., 1.], [1., 0.]]); A
array([[0., 1.],
       [1., 0.]])
>>> I * A
array([[0., 0.],
       [0., 0.]])
>>> I @ A
array([[0., 1.],
       [1., 0.]])
>>> x = np.array([1., 2.]); x
array([1., 2.])
>>> A @ x
array([2., 1.])
